Day 22 处理字符串的函数

之前我们讲过 C 语言的 字符串。

C 语言以字符的数组来表示字符串，对此标准库提供了一些专门用于对字符串进行处理的函数。使用这些函数之前，必须引入 string.h 头文件。

strlen 函数用于计算字符串的长度

    #include <stdio.h>
    #include <string.h>
    
    int main() {
        char a [101];
        printf("please input a string (no more than 100 chars)\n");
        scanf("%s", a);
        printf("length of \"%s\" is %d.\n", a, strlen(a));
        return 0;
    }

strchr 函数用于在字符串中查找某个字符

            
    #include <stdio.h>
    #include <string.h>
    
    int main() {
        char a[101], c;
        printf("please input a string and a char\n");
        scanf("%s %c", a, &c);
        char * p = strchr(a, c);
        if (p != NULL) {
            printf("%d\n", p - a);
        } else {
            printf("\"%s\" does not contain '%c'\n", a, c);
        }
        return 0;
    }

strpbrk(const char * s1, const char * s2) 在字符串 s1 中查找字符串 s2 中的任意字符

            
    #include <stdio.h>
    #include <string.h>
    
    int main() {
        char a[101], b[101];
        printf("please input two strings\n");
        scanf("%s %s", a, b);
        char * p = strpbrk(a, b);
        if (p != NULL) {
            printf("%d\n", p - a);
        } else {
            printf("\"%s\" does not contain any char in \"%s\"\n", a, b);
        }
        return 0;
    }

                            你把所有在初始化时忘记设置长度的全都设置好长度
                            
char s[] = "123";

                            这种代码很危险
                            

                            在声明 s 的时候, 没有指明长度, s 的长度默认是 4, 因为字符串最后有一个 '\0'
                            

                            如果用 strcpy 等方法给 s 赋值更长的字符串, 就会发生内存溢出错误
                            

                            我们在写代码时候, [] 里面不指明长度也可以, 但是最好这样写
                            
const char s[] = "123";

                            这样字符串初始化之后就不能被修改了
