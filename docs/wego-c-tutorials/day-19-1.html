递归

递归调用即函数调用本身，递归会有一个终止条件。所以递归函数一定包含分支结构。

利用递归求 1 到 10 的阶乘的代码如下：

    #include <stdio.h>

    int factory(int n) {
        if (n == 1 || n == 0) return 1;
        else return factory(n-1) * n;
    }

    int main() {
        int n;
        for (n = 1; n <= 10; ++n) {
            printf("%2d! = %8d\n", n, factory(n));
        }
        return 0;
    }

利用递归求 Fibonacci 数列 前40个数的代码如下：

    #include <stdio.h>

    int Fibonacci(int n) {
        if (n == 1 || n == 2) return 1;
        else return Fibonacci(n - 1) + Fibonacci(n - 2);
    }

    int main() {
        int n;
        for (n = 1; n <= 40; ++n) {
            printf(" %10d", Fibonacci(n));
            if (n % 10 == 0) printf("\n");
        }
        return 0;
    }

可以看出，当递归层数很多时，程序的执行时间很长。利用动态规划的方法可以加快程序执行的时间：

            

            
            
    #include <stdio.h>
    #define SIZE 50

    int fab(int n, int m[]) {
        if (m[n]) {
            return m[n];
        } else {
            m[n] = fab(n - 1, m) + fab(n - 2, m);
            return m[n];
        }
    }

    int memFab(int n) {
        int m[SIZE];
        int i;
        m[0] = 1;
        m[1] = 1;
        for(i = 2; i < SIZE; ++i) {
            m[i] = 0;
        }
        return fab(n, m);
    }

    int main() {
        int n;
        for (n = 1; n <= 40; ++n) {
            printf(" %10d", memFab(n));
            if (n % 10 == 0) printf("\n");
        }
        return 0;
    }

值得一提的是，主函数也可以递归。

                
下面的程序运用主函数递归输出九九乘法表。

            

            
            
    #include <stdio.h>

    int main(int argc, char ** argv) {
        if (argc == 10) {
            printf("%28s","九九乘法表");
        } else if (argc % 10 == 1 || argc <= 11) {
        } else {
            if (argc % 10 == 2) printf("|");
            if (argc <= 20) {
                printf("%4d|", (argc - 1) % 10);
            } else {
                if (argc % 10 == 0) {
                    printf("%4d|", (argc/10 - 1) * ((argc-1) % 10));
                } else {
                    printf("%4d|", (argc/10) * ((argc-1) % 10));
                }
            }
        }
        if (argc % 10 == 0) {
            printf("\n+----+----+----+----+----+----+----+----+----+\n");
        }
        if (argc == 100) {
            return 0;
        }
        return main(++argc, argv);
    }
