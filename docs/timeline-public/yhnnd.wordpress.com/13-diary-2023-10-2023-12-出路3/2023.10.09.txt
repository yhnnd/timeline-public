function isFinal (s) {
    if (s.length == 0) {
        return true;
    }
    let pos = 0, neg = 0;
    for(let i of s) {
        if (i > 0) {
            pos++;
        } else if (i < 0) {
            neg++;
        }
    }
    if (pos == 0 || neg == 0) {
        return true;
    }
    return false;
}

function collide (s) {
    let ns = [];
    for (i = 0; i < s.length; ++i) {
        if (s[i] > 0) {
            if (i + 1 < s.length && s[i + 1] < 0) {
                if (s[i] > - s[i + 1]) {
                    ns.push(s[i]);
                } else if (s[i] < - s[i + 1]) {
                    ns.push(s[i + 1]);
                } else {
                }
                ++i;
            } else {
                ns.push(s[i]);
            }
        } else {
            ns.push(s[i]);
        }
    }
    return ns;
}

function calc (s) {
    while (isFinal(s) == false) {
        let ns = collide (s);
        if (JSON.stringify(ns) == JSON.stringify(s)) {
            break;
        }
        s = ns;
    }
