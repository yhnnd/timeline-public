#include <stdio.h>
#define CHARMAX 128
void putline(char *line,FILE *fp){
	int i;
	static bool flag=1,flag2=1;
	for(i=0;line[i]!=0&&line[i]!=EOF;i++){
		if(line[i]=='/'&&line[i+1]=='*') flag=0;
		if(line[i]=='/'&&line[i+1]=='/') flag2=0;
		if(line[i]=='\n') flag2=1;
		if(flag&&flag2) fputc(line[i],fp);
		if(line[i]=='*'&&line[i+1]=='/'){
			flag=1;
			i++;
		}
	}
}
int main(){
	char line[CHARMAX];
	FILE *fp=fopen("src.c","r"),*fp2=fopen("dst.c","w");
	for(;;){
		fgets(line,CHARMAX,fp);
		if(feof(fp)) break;
		putline(line,fp2);
	}
	fclose(fp);
	fclose(fp2);
}
